function enviarReserva() { if (!mesaSeleccionada) { alert("Selecciona una mesa"); return; } const datos = { nombre: document.getElementById('nombre').value, fecha: document.getElementById('fecha').value, mesa: mesaSeleccionada, personas: document.getElementById('personas').value, platos: Array.from(document.querySelectorAll('.check-plato:checked')).map(p => p.value) }; fetch('https://restaurante-raiz.onrender.com/reservar', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify(datos) }) .then(res => res.json()) .then(data => { alert("¡Reserva enviada! Espera nuestra confirmación."); location.reload(); // Limpia la página }) .catch(err => alert("Error al enviar")); }
